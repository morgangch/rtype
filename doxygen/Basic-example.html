ECS Usage

#include <ECS/ECS.h>
// Define a component
class Position : public ECS::Component<Position> {
public:
float x, y;
Position(float x = 0, float y = 0) : x(x), y(y) {}
};
// Use the ECS
ECS::World world;
auto entity = world.CreateEntity();
world.AddComponent<Position>(entity, 10.0f, 20.0f);
auto pos = world.GetComponent<Position>(entity);
