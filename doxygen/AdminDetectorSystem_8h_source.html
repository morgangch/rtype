45 void Update(ECS::World &world, float deltaTime) override {

52 // Check if the admin player is still connected on that room
53 bool admin_still_connected = false;
54 for (auto &player_pair: *world.GetAllComponents<rtype::server::components::PlayerConn>()) {
55 ECS::EntityID playerId = player_pair.first;
56 rtype::server::components::PlayerConn *playerConn = player_pair.second.get();
57 if (!playerConn)
58 continue;
59 if (playerConn->room_code == room->joinCode && playerId == admin_id) {
60 admin_still_connected = true;
61 break;
62 }
63 }
64 if (admin_still_connected)
65 continue;
66 // Admin has disconnected, assign a new admin
67 for (auto &conn_pair: *world.GetAllComponents<rtype::server::components::PlayerConn>()) {
68 rtype::server::components::PlayerConn *playerConn = conn_pair.second.get();
69 if (!playerConn)
70 continue;
71 if (playerConn->room_code != room->joinCode)
72 continue;
73 // Assign new admin
74 room->ownerId = conn_pair.first;
75 RoomAdminUpdatePacket p{conn_pair.first};
76 // Broadcast to all players in the room about the new admin
77 room->broadcastPacket(&p, sizeof(p), ROOM_ADMIN_UPDATE, true);
78 break;
79 }
